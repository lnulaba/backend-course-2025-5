# Команди для тестування проксі-сервера

## Запуск сервера
node server.js --host localhost --port 3000 --cache ./cache

## Або для розробки з nodemon
npm run dev -- --host localhost --port 3000 --cache ./cache

## Тестування GET запиту (отримання картинки)
curl -X GET http://localhost:3000/200 --output test_200.jpg

## Тестування GET запиту для неіснуючого коду
curl -X GET http://localhost:3000/999

## Тестування PUT запиту (збереження картинки)
curl -X PUT http://localhost:3000/201 --data-binary @test_image.jpg

## Тестування DELETE запиту (видалення картинки з кешу)
curl -X DELETE http://localhost:3000/200

## Тестування невідомого методу
curl -X PATCH http://localhost:3000/200

## Тестування з браузером
# Відкрийте в браузері: http://localhost:3000/200
